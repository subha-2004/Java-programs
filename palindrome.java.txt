import java.util.Scanner;
class Palindrome 
{
	static Scanner sc=new Scanner(	System.in);
	public static void main(String[] args) 
	{
		System.out.println("Enter one number");
		int num=sc.nextInt();
		int rev=0;
		int temp=num;
		while(num>0)
		{
			int ld=num%10;      
			rev=rev*10+ld;
			num=num/10;
		}
		if(temp==rev)
		{
			System.out.println("palindrome");
		}
		else
		{
			System.out.println("not a palindrome");
		}
	}
}
